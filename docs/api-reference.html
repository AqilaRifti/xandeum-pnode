API Reference

Server functions and utilities for working with pNode data.

Data Functions

getDashboardData

Fetches and processes all pNode data for the dashboard.

// lib/nodeData.server.ts
export async function getDashboardData(): Promise<DashboardData>

// Returns
interface DashboardData {
  nodes: PNode[];
  stats: NetworkStats;
  versionDistribution: VersionData[];
  storageMetrics: StorageMetrics;
}

getNodeById

Fetches a single pNode by its public key.

export async function getNodeById(pubkey: string): Promise<PNode | null>

// Usage
const node = await getNodeById('EcTqXgB6VJStAtBZAXcjLHf5ULj41H1PFZQ17zKosbhL');

getNodesByStatus

Fetches nodes filtered by status.

export async function getNodesByStatus(
  status: 'online' | 'offline'
): Promise<PNode[]>

// Usage
const onlineNodes = await getNodesByStatus('online');

getNodesByVersion

Fetches nodes filtered by software version.

export async function getNodesByVersion(version: string): Promise<PNode[]>

// Usage
const v8Nodes = await getNodesByVersion('0.8.0');

Type Definitions

PNode

interface PNode {
  pubkey: string;
  status: 'online' | 'offline';
  version: string;
  storageUsed: number;
  storageTotal: number;
  storageCommitted: number;
  storageUsagePercent: number;
  uptime: number;
  ip: string;
  address: string;
  isPublic: boolean;
  rpcPort: number;
  lastSeen: string;
  lastSeenTimestamp: number;
  ramUsed?: number;
  ramTotal?: number;
}

NetworkStats

interface NetworkStats {
  totalNodes: number;
  onlineNodes: number;
  offlineNodes: number;
  totalStorage: number;
  usedStorage: number;
  averageUptime: number;
  publicNodes: number;
  privateNodes: number;
}

VersionData

interface VersionData {
  version: string;
  count: number;
  percentage: number;
}

StorageMetrics

interface StorageMetrics {
  totalCapacity: number;
  usedCapacity: number;
  committedCapacity: number;
  availableCapacity: number;
  utilizationPercent: number;
}

// lib/format.ts
export function formatStorage(bytes: number): string

// Usage

Formats uptime in days to human-readable format.

export function formatUptime(days: number): string
