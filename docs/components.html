Components

Key components used in Xandeum pNode Analytics for displaying node data and network metrics.

Dashboard Components

DashboardClient

The main client component that renders the dashboard with pNode data:

import { DashboardClient } from '@/components/dashboard/DashboardClient';

// Used in page.tsx with server-fetched data
<DashboardClient initialData={data} />

Stats Cards

Display key network metrics:

import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';

function StatsCard({ title, value, description, icon }) {
  return (
    <Card>
      <CardHeader className="flex flex-row items-center justify-between">
        <CardTitle className="text-sm font-medium">{title}</CardTitle>
        {icon}
      </CardHeader>
      <CardContent>
        <div className="text-2xl font-bold">{value}</div>
        <p className="text-xs text-muted-foreground">{description}</p>
      </CardContent>
    </Card>
  );
}

// Usage
<StatsCard
  title="Total Nodes"
  value={stats.total}
  description={`${stats.online} online, ${stats.offline} offline`}
  icon={<Server className="h-4 w-4" />}
/>

Data Table

Display pNode data with sorting, filtering, and pagination:

import { DataTable } from '@/components/ui/data-table';

// Define columns for pNode data
const columns = [
  {
    accessorKey: 'pubkey',
    header: 'Public Key',
    cell: ({ row }) => (
      <code className="text-xs">
        {row.original.pubkey.slice(0, 8)}...
      </code>
    ),
  },
  {
    accessorKey: 'status',
    header: 'Status',
    cell: ({ row }) => (
      <Badge variant={row.original.status === 'online' ? 'success' : 'destructive'}>
        {row.original.status}
      </Badge>
    ),
  },
  {
    accessorKey: 'version',
    header: 'Version',
  },
  {
    accessorKey: 'storageUsagePercent',
    header: 'Storage',
    cell: ({ row }) => (
      <Progress value={row.original.storageUsagePercent * 100} />
    ),
  },
  {
    accessorKey: 'uptime',
    header: 'Uptime',
    cell: ({ row }) => `${row.original.uptime.toFixed(1)} days`,
  },
];

// Usage
<DataTable columns={columns} data={nodes} />

Charts

Recharts-powered visualizations:

Storage Distribution Chart

import { PieChart, Pie, Cell, ResponsiveContainer, Legend } from 'recharts';

function StorageChart({ data }) {
  return (
    <ResponsiveContainer width="100%" height={300}>
      <PieChart>
        <Pie
          data={data}
          dataKey="value"
          nameKey="name"
          cx="50%"
          cy="50%"
          outerRadius={80}
        >
          {data.map((entry, index) => (
            <Cell key={index} fill={COLORS[index % COLORS.length]} />
          ))}
        </Pie>
        <Legend />
      </PieChart>
    </ResponsiveContainer>
  );
}

Version Distribution Chart

import { BarChart, Bar, XAxis, YAxis, Tooltip, ResponsiveContainer } from 'recharts';

function VersionChart({ versionData }) {
  return (
    <ResponsiveContainer width="100%" height={300}>
      <BarChart data={versionData}>
        <XAxis dataKey="version" />
        <YAxis />
        <Tooltip />
        <Bar dataKey="count" fill="#3b82f6" />
      </BarChart>
    </ResponsiveContainer>
  );
}

Map Component

Interactive map showing node locations:

import { MapContainer, TileLayer, Marker, Popup } from 'react-leaflet';

function NodeMap({ nodes }) {
  return (
    <MapContainer
      center={[20, 0]}
      zoom={2}
      className="h-[400px] w-full rounded-lg"
    >
      <TileLayer
        url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png"
        attribution='&copy; OpenStreetMap'
      />
      {nodes.map((node) => (
        <Marker key={node.pubkey} position={[node.lat, node.lng]}>
          <Popup>
            <div>
              <strong>{node.pubkey.slice(0, 8)}...</strong>
              <p>Status: {node.status}</p>
              <p>Version: {node.version}</p>
            </div>
          </Popup>
        </Marker>
      ))}
    </MapContainer>
  );
}

Adding New Components

Add Shadcn UI components using the CLI:

# Add a component
npx shadcn@latest add [component-name]

# Examples
npx shadcn@latest add card
npx shadcn@latest add table
npx shadcn@latest add chart
